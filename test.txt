boo := 3; //staticly defined variable

func main() {
    bar := 9;

    print(boo); //compiles to PRIA @var_stat_boo

    print(bar); //compiles to RMOV @var_dyn_bar @temp_0
                //            PRIA @temp_0

    boo = bar;   //compiles to RMOV @var_dyn_bar @var_stat_boo
    bar = boo;   //compiles to MOVR @var_stat_boo @var_dyn_bar

    for (i := 0; i < 10; i++) {
        if (i%2 == 0){
            print("odd");
        } else {
            print("even");
        }
        print(i+1);
    }

    j := 10;
    while (j > 0) {
        print(j);
        j--;
    }
}
