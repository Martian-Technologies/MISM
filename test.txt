func main() {
    for (i := 0; i < 10; i++) {
        if (i%2 == 0){
            print("odd");
        } else {
            print("even");
        }
        print(i+1);
    }

    j := 10;
    while (j > 0) {
        print(j);
        j--;
    }
}

//[['func', 'main', [], [['for', [['i', ':=', '0'], ['i', '<', '10'], ['i', '++']], [['print', [['i', '+', '1']]]]], ['j', ':=', '10'], ['while', [['j', '>', '0']], [['print', [['j']]], ['j', '--']]]]]]
